00153 /* RFC 2631, Section 2.1.5, http://www.ietf.org/rfc/rfc2631.txt */
00154 static int
00155 isValidPublicKey(MP_t y, MP_t p, MP_t q)
00156 {
00157   int ret = TRUE;
00158   MP_t bn;
00159   assert(y);
00160 
00161   MP_new(bn);
00162   assert(bn);
00163 
00164   /* y must lie in [2,p-1] */
00165   MP_set_w(bn, 1);
00166   if (MP_cmp(y, bn) < 0)
00167     {
00168       RTMP_Log(RTMP_LOGERROR, "DH public key must be at least 2");
00169       ret = FALSE;
00170       goto failed;
00171     }
00172 
00173   /* bn = p-2 */
00174   MP_set(bn, p);
00175   MP_sub_w(bn, 1);
00176   if (MP_cmp(y, bn) > 0)
00177     {
00178       RTMP_Log(RTMP_LOGERROR, "DH public key must be at most p-2");
00179       ret = FALSE;
00180       goto failed;
00181     }
00182 
00183   /* Verify with Sophie-Germain prime
00184    *
00185    * This is a nice test to make sure the public key position is calculated
00186    * correctly. This test will fail in about 50% of the cases if applied to
00187    * random data.
00188    */
00189   if (q)
00190     {
00191       /* y must fulfill y^q mod p = 1 */
00192       MP_modexp(bn, y, q, p);
00193 
00194       if (MP_cmp_1(bn) != 0)
00195         {
00196           RTMP_Log(RTMP_LOGWARNING, "DH public key does not fulfill y^q mod p = 1");
00197         }
00198     }
00199 
00200 failed:
00201   MP_free(bn);
00202   return ret;
00203 }

00240   size_t res = 0;
