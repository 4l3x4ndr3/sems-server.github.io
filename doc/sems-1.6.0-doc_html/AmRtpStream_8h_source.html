00103 class PayloadMask
00104 {
00105   private:
00106     unsigned char bits[16];
00107 
00108   public:
00109     // clear flag for all payloads
00110     void clear();
00111 
00112     // set given flag (TODO: once it shows to be working, change / and % to >> and &)
00113     void set(unsigned char payload_id) { if (payload_id < 128) bits[payload_id / 8] |= 1 << (payload_id % 8); }
00114 
00115     // set all flags to 'true'
00116     void set_all();
00117 
00118     // invert all flags
00119     void invert();
00120 
00121     // get given flag
00122     bool get(unsigned char payload_id) { if (payload_id > 127) { ERROR("BUG: payload_id out of range\n"); return false; } return (bits[payload_id / 8] & (1 << (payload_id % 8))); }
00123     
00124     PayloadMask() { clear(); }
00125     PayloadMask(bool _set_all) { if (_set_all) set_all(); else clear(); }
