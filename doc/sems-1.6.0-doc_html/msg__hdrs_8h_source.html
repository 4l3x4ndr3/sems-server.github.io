00087 inline void via_wr(char** c, const cstring& trsp, const cstring& addr, 
00088                    const cstring& branch, bool rport)
00089 {
00090     memcpy(*c,"Via: SIP/2.0/",13);
00091     *c += 13/*'Via: SIP/2.0/'*/;
00092 
00093     for(unsigned int i=0; i<trsp.len; i++) {
00094       if(trsp.s[i] >= 'a' && trsp.s[i] <= 'z')
00095         *((*c)++) = trsp.s[i] - 'a' + 'A';
00096     }
00097     
00098     *((*c)++) = SP;
00099 
00100     memcpy(*c,addr.s,addr.len);
00101     *c += addr.len;
00102 
00103     memcpy(*c,";branch=" MAGIC_BRANCH_COOKIE,8+MAGIC_BRANCH_LEN);
00104     *c += 8+MAGIC_BRANCH_LEN;
00105 
00106     memcpy(*c,branch.s,branch.len);
00107     *c += branch.len;
00108     
00109     if(rport){
00110       memcpy(*c,";rport",6);
00111       *c += 6;
00112     }
00113 
00114     *((*c)++) = CR;
