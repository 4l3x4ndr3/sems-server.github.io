00060 // Wallclock definitions:
00061 //
00062 // The wallclock is defined such that:
00063 //  - it is the highest clock rate is the system
00064 //  - any supported sample rate must be smaller
00065 //  - the difference between scaled down timers
00066 //    is always consistent with respect to overflows.
00067 //  - supported sample rates are multiples of 100
00068 //    (44100 is supported, 44110 is not)
00069 #define WALLCLOCK_RATE 102400LL
00070 //
00071 // Wallclock overflow mask
00072 #define WALLCLOCK_MASK 0xFFFFFFFFFFFFLL // 48 bit mask
00073 //
00074 // Wallclock increments
00075 #define WC_INC_MS 10LL /* 10 ms */
00076 #define WC_INC ((WALLCLOCK_RATE*WC_INC_MS)/1000LL)
