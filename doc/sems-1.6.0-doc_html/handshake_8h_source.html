00276 static void
00277 CalculateDigest(unsigned int digestPos, uint8_t *handshakeMessage,
00278                 const uint8_t *key, size_t keyLen, uint8_t *digest)
00279 {
00280   const int messageLen = RTMP_SIG_SIZE - SHA256_DIGEST_LENGTH;
00281   uint8_t message[RTMP_SIG_SIZE - SHA256_DIGEST_LENGTH];
00282 
00283   memcpy(message, handshakeMessage, digestPos);
00284   memcpy(message + digestPos,
00285          &handshakeMessage[digestPos + SHA256_DIGEST_LENGTH],
00286          messageLen - digestPos);
00287 
00288   HMACsha256(message, messageLen, key, keyLen, digest);
00289 }

00332 /* RTMPE type 8 uses XTEA on the regular signature
00333  * http://en.wikipedia.org/wiki/XTEA
00334  */
00335 static void rtmpe8_sig(uint8_t *in, uint8_t *out, int keyid)
00336 {
00337   unsigned int i, num_rounds = 32;
00338   uint32_t v0, v1, sum=0, delta=0x9E3779B9;
00339   uint32_t const *k;
00340 
00341   v0 = in[0] | (in[1] << 8) | (in[2] << 16) | (in[3] << 24);
00342   v1 = in[4] | (in[5] << 8) | (in[6] << 16) | (in[7] << 24);
00343   k = rtmpe8_keys[keyid];
00344 
00345   for (i=0; i < num_rounds; i++) {
00346     v0 += (((v1 << 4) ^ (v1 >> 5)) + v1) ^ (sum + k[sum & 3]);
00347     sum += delta;
00348     v1 += (((v0 << 4) ^ (v0 >> 5)) + v0) ^ (sum + k[(sum>>11) & 3]);
00349   }
00350 
00351   out[0] = v0; v0 >>= 8;
00352   out[1] = v0; v0 >>= 8;
00353   out[2] = v0; v0 >>= 8;
00354   out[3] = v0;
00355 
00356   out[4] = v1; v1 >>= 8;
00357   out[5] = v1; v1 >>= 8;
00358   out[6] = v1; v1 >>= 8;
00359   out[7] = v1;
00360 }
