00164 inline int lower_cmp(const char* l, const char* r, int len)
00165 {
00166     const char* end = l+len;
00167 
00168     while(l!=end){
00169         if( LOWER_B(*l) == LOWER_B(*r) ){
00170             l++; r++;
00171             continue;
00172         }
00173         else if(LOWER_B(*l) < LOWER_B(*r)) {
00174             return -1;
00175         }
00176         else {
00177             return 1;
00178         }
00179     }

00181     return 0;
